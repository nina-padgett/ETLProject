CREATE TABLE STATE_COVID_19_ACTIVITY (
	PROVINCE_STATE_NAME VARCHAR PRIMARY KEY,
	PEOPLE_POSITIVE_CASES_COUNT BIGINT,
	PEOPLE_DEATH_NEW_COUNT INT,
	PEOPLE_POSITIVE_NEW_CASES_COUNT INT,
	PEOPLE_DEATH_COUNT INT
);

CREATE TABLE US_POPULATION (
	PROVINCE_STATE_NAME VARCHAR(20) PRIMARY KEY,
	POPULATION INT
);

SELECT state_covid_19_activity.PROVINCE_STATE_NAME, state_covid_19_activity.PEOPLE_POSITIVE_CASES_COUNT, state_covid_19_activity.PEOPLE_DEATH_NEW_COUNT, state_covid_19_activity.PEOPLE_POSITIVE_NEW_CASES_COUNT, state_covid_19_activity.PEOPLE_DEATH_COUNT
FROM state_covid_19_activity
JOIN us_population
ON state_covid_19_activity.PROVINCE_STATE_NAME = us_population.PROVINCE_STATE_NAME;
